let pub = null;

function init() {
  const stored = localStorage.getItem('inkstain_pub');
  if (!stored) { window.location.href = '/publishers'; return; }
  pub = JSON.parse(stored);
  document.getElementById('pub-name').textContent = pub.name;
  document.getElementById('pol-require').checked = pub.policy_requires_trail || false;
  document.getElementById('pol-disclosure').value = pub.policy_disclosure_level || 'summary';
  document.getElementById('pol-ai').value = pub.policy_accepts_ai || 'case_by_case';
  const authors = (pub.authors||[]).length;
  document.getElementById('stat-authors').textContent = authors;
  document.getElementById('stat-certs').textContent = '0';
  document.getElementById('stat-arr').textContent = authors > 0 ? '$'+(authors*199).toLocaleString() : '$0';
  loadInviteLink();
}

async function loadInviteLink() {
  const r = await fetch(`/api/publishers/invite?id=${pub.id}`);
  const d = await r.json();
  document.getElementById('invite-link').textContent = d.invite_link || 'Error loading link';
}

async function savePolicy() {
  const r = await fetch('/api/publishers/policy', {
    method:'POST', headers:{'Content-Type':'application/json'},
    body: JSON.stringify({
      id: pub.id, email: pub.email,
      requires_trail: document.getElementById('pol-require').checked,
      disclosure_level: document.getElementById('pol-disclosure').value,
      accepts_ai: document.getElementById('pol-ai').value
    })
  });
  const d = await r.json();
  if (d.ok) {
    const ok = document.getElementById('pol-ok');
    ok.style.display = 'block';
    setTimeout(()=>ok.style.display='none', 3000);
  }
}

async function copyInvite() {
  const link = document.getElementById('invite-link').textContent;
  await navigator.clipboard.writeText(link);
  const ok = document.getElementById('copy-ok');
  ok.style.display = 'block';
  setTimeout(()=>ok.style.display='none', 2000);
}

async function verifyCert() {
  const hash = document.getElementById('verify-hash').value.trim();
  if (!hash) return;
  const r = await fetch(`/api/verify?hash=${encodeURIComponent(hash)}`);
  const d = await r.json();
  const el = document.getElementById('verify-result');
  if (d.verified) {
    const t = d.trail_summary || {};
    el.innerHTML = `<div class="result-ok">
      <div class="badge">✦ Verified</div>
      <div style="font-family:'Playfair Display',serif;font-size:18px;margin-bottom:.25rem;">"${esc(d.title)}"</div>
      <div style="font-size:13px;color:rgba(245,242,235,.6);margin-bottom:1rem;">by ${esc(d.author)} · ${esc(d.generated_at)}</div>
      <div style="display:grid;grid-template-columns:repeat(4,1fr);gap:.5rem;margin-bottom:1rem;">
        ${[['AI copies',t.ai_copies||0],['AI sessions',t.ai_sessions||0],['Write cycles',t.interleave_count||0],['Min in AI',t.ai_active_minutes||0]].map(([l,v])=>`<div style="background:rgba(245,242,235,.06);padding:.75rem;"><div style="font-size:20px;font-weight:700;">${v}</div><div style="font-size:10px;color:rgba(245,242,235,.4);letter-spacing:1px;text-transform:uppercase;">${l}</div></div>`).join('')}
      </div>
      <div style="font-size:12px;color:rgba(245,242,235,.35);">Disclosure: ${d.disclosure_level||'—'} · Trail captured ${esc(d.generated_at||'')}</div>
      ${d.author_note?`<div style="font-size:13px;color:rgba(245,242,235,.5);margin-top:.75rem;font-style:italic;">"${esc(d.author_note)}"</div>`:''}
    </div>`;
  } else {
    el.innerHTML = `<div class="result-fail">Certificate not found. Double-check the hash from the bottom of the PDF.</div>`;
  }
}

function esc(s) { return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }
function signout() { localStorage.removeItem('inkstain_pub'); window.location.href='/publishers'; }

init();

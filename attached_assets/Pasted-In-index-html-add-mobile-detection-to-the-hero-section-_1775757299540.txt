async function sendDownloadLink() {
  const email = document.getElementById('mobile-email').value.trim();
  const status = document.getElementById('mobile-email-status');
  if (!email || !email.includes('@')) {
    status.textContent = 'Please enter a valid email address.';
    status.style.color = '#8b3020';
    return;
  }
  status.textContent = 'Sending...';
  status.style.color = 'var(--muted)';
  try {
    const res = await fetch('/api/sendlink', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({email})
    });
    const data = await res.json();
    if (data.ok) {
      status.textContent = '✦ Link sent. Check your email when you\'re at your computer.';
      status.style.color = 'var(--amber)';
      document.getElementById('mobile-email').value = '';
    } else {
      status.textContent = 'Something went wrong. Try again.';
      status.style.color = '#8b3020';
    }
  } catch(err) {
    status.textContent = 'Could not connect. Try again.';
    status.style.color = '#8b3020';
  }
}
</script>
